// ───────────────────────────────────────────── // Citation Manager — Google Apps Script // Code.gs v3 — Two-step search + parseRefText fallback // ───────────────────────────────────────────── function doGet(e) { return HtmlService.createHtmlOutputFromFile('index') .setTitle('Citation Manager — Tech Economist') .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL); } // ── SCORE metadata quality ──────────────────── function scoreResult(m) { if (!m) return 0; var s = 0; if (m.authors && m.authors.length > 0) s += 30; if (m.year) s += 15; if (m.title) s += 15; if (m.venue) s += 15; if (m.vol) s += 8; if (m.iss) s += 5; if (m.pages) s += 7; if (m.doi) s += 5; return s; } // ── PARSE OpenAlex response ─────────────────── function parseOA(w) { var authors = (w.authorships || []).map(function(a) { var name = (a.author && a.author.display_name) ? a.author.display_name : ''; var parts = name.trim().split(/\s+/); var last = parts.length > 1 ? parts[parts.length - 1] : name; var first = parts.length > 1 ? parts.slice(0, -1).join(' ') : ''; var initials = first ? first.split(/\s+/).map(function(p){ return p[0] ? p[0].toUpperCase() + '.' : ''; }).join(' ') : ''; return { last: last, first: first, initials: initials, full: name }; }); var biblio = w.biblio || {}; var pages = (biblio.first_page && biblio.last_page) ? biblio.first_page + '\u2013' + biblio.last_page : (biblio.first_page || ''); var venue = ''; if (w.primary_location && w.primary_location.source) venue = w.primary_location.source.display_name || ''; if (!venue && w.best_oa_location && w.best_oa_location.source) venue = w.best_oa_location.source.display_name || ''; return { authors: authors, title: w.title || '', year: w.publication_year || '', venue: venue, vol: biblio.volume || '', iss: biblio.issue || '', pages: pages, doi: w.doi ? w.doi.replace('https://doi.org/', '') : '', source: 'OpenAlex', step: 2 }; } // ── PARSE CrossRef response ─────────────────── function parseCR(w) { var authors = (w.author || []).map(function(a) { var first = a.given || '', last = a.family || ''; var initials = first ? first.split(/\s+/).map(function(p){ return p[0] ? p[0].toUpperCase() + '.' : ''; }).join(' ') : ''; return { last: last, first: first, initials: initials, full: (first + ' ' + last).trim() }; }); var page = w.page || '', pageParts = page.split(/[-\u2013]/); var pages = (pageParts[0] && pageParts[1]) ? pageParts[0] + '\u2013' + pageParts[1] : page; var year = ''; if (w.published && w.published['date-parts'] && w.published['date-parts'][0]) year = w.published['date-parts'][0][0]; if (!year && w['published-print'] && w['published-print']['date-parts'] && w['published-print']['date-parts'][0]) year = w['published-print']['date-parts'][0][0]; return { authors: authors, title: (w.title && w.title[0]) ? w.title[0] : '', year: year || '', venue: (w['container-title'] && w['container-title'][0]) ? w['container-title'][0] : '', vol: w.volume || '', iss: w.issue || '', pages: pages, doi: w.DOI || '', source: 'CrossRef', step: 2 }; } // ── PARSE reference text directly (fallback) ── function parseRefText(ref) { var doiM = ref.match(/(?:https?:\/\/doi\.org\/|doi:\s*)(10\.[^\s,\]>"'{}\\]+)/i); var doi = doiM ? doiM[1].replace(/[.)\s]+$/, '') : ''; var yearM = ref.match(/\b(19|20)\d{2}\b/); var year = yearM ? yearM[0] : 'n.d.'; var yearPos = yearM ? yearM.index : ref.length; var authorBlock = ref.substring(0, yearPos).trim().replace(/[.,( ]+$/, ''); authorBlock = authorBlock.replace(/\s*[&]\s*/g, ', ').replace(/\s+and\s+/gi, ', '); var authors = []; var authorRe = /([A-Z][a-z\u00C0-\u024F'\-]+(?:\s+[A-Z][a-z\u00C0-\u024F'\-]+)*),?\s+([A-Z][.\s]*(?:[A-Z]\.?\s*)*)(?=[,;]|$)/g; var am; while ((am = authorRe.exec(authorBlock)) !== null) { var last = am[1].trim(); var rawIni = am[2].trim(); var iniLetters = rawIni.match(/[A-Z]/g) || []; var initials = iniLetters.join('.') + (iniLetters.length ? '.' : ''); authors.push({ last: last, first: rawIni, initials: initials, full: last + ', ' + initials }); } if (!authors.length) { var fb = authorBlock.split(',')[0].trim(); if (fb) authors.push({ last: fb, first: '', initials: '', full: fb }); } var afterYear = ref.substring(yearPos + year.length).replace(/^[\s).]+/, ''); var titleM = afterYear.match(/^([A-Z*\[][^.]{5,180}?)\./); var title = titleM ? titleM[1].trim() : afterYear.split('.')[0].trim(); title = title.replace(/\s*\([^)]*(?:ed|edition|edn)[^)]*\)/i, '').trim(); var afterTitle = afterYear.substring(title.length + 1).replace(/https?:\/\/\S+/g, '').trim().replace(/[.,]+$/, ''); var pubM = afterTitle.match(/^([^:,.\n]+)/); var publisher = pubM ? pubM[1].trim() : ''; return { authors: authors, year: year, title: title, venue: publisher, vol: '', iss: '', pages: '', doi: doi, source: 'Parsed', step: 0 }; } // ── EXTRACT helpers ─────────────────────────── function extractDOI(ref) { var m = ref.match(/(?:https?:\/\/doi\.org\/|doi:\s*)(10\.[^\s,\]>"'{}\\]+)/i); if (m) return m[1].replace(/[.)\s]+$/, ''); var m2 = ref.match(/\b(10\.\d{4,}\/[^\s,\]>"'{}\\]+)/); return m2 ? m2[1].replace(/[.)\s]+$/, '') : null; } function extractFirstAuthorLast(ref) { var m = ref.match(/^([A-Z][a-zA-Z'\u00C0-\u024F\-]{2,})/); return m ? m[1].toLowerCase().replace(/[^a-z]/g, '') : ''; } function extractRefYear(ref) { var m = ref.match(/\b(19|20)\d{2}\b/); return m ? parseInt(m[0]) : null; } // ── VERIFY match ────────────────────────────── function verifyMatch(meta, origAuthor, origYear) { if (!meta) return false; if (origYear && meta.year && Math.abs(parseInt(meta.year) - origYear) > 2) return false; if (origAuthor && origAuthor.length > 3 && meta.authors && meta.authors.length > 0) { var metaLast = meta.authors[0].last.toLowerCase().replace(/[^a-z]/g, ''); if (metaLast.length > 3) { var a4 = origAuthor.substring(0, 4), b4 = metaLast.substring(0, 4); if (metaLast.indexOf(a4) === -1 && origAuthor.indexOf(b4) === -1) return false; } } return true; } // ── DOI direct lookup ───────────────────────── function lookupByDOI(doi, origAuthor, origYear, refText) { try { var url = 'https://api.crossref.org/works/' + encodeURIComponent(doi) + '?mailto=techeconomist@research.org'; var res = UrlFetchApp.fetch(url, { muteHttpExceptions: true }); if (res.getResponseCode() === 200) { var data = JSON.parse(res.getContentText()); if (data.message) { var m = parseCR(data.message); m.step = 1; // If author/year mismatch (e.g. CrossRef returned a book review) — parse text directly if (!verifyMatch(m, origAuthor, origYear)) { var parsed = parseRefText(refText); parsed.doi = doi; parsed.step = 1; parsed.source = 'CrossRef+Parsed'; return parsed; } return m; } } } catch(e) { /* silent */ } return null; } // ── MAIN: called from browser ───────────────── function searchAPIs(refText, title) { var origAuthor = extractFirstAuthorLast(refText || ''); var origYear = extractRefYear(refText || ''); var doi = extractDOI(refText || ''); // Strategy 1: DOI lookup if (doi) { var doiMeta = lookupByDOI(doi, origAuthor, origYear, refText); if (doiMeta) return doiMeta; } // Strategy 2: Title + author + year verified search var searchTitle = (title || refText || '').substring(0, 130); var oaMeta = null, crMeta = null; try { var oaUrl = 'https://api.openalex.org/works?search=' + encodeURIComponent(searchTitle) + '&per-page=5&mailto=techeconomist@research.org'; var oaRes = UrlFetchApp.fetch(oaUrl, { muteHttpExceptions: true }); if (oaRes.getResponseCode() === 200) { var oaData = JSON.parse(oaRes.getContentText()); var items = oaData.results || []; for (var i = 0; i < items.length; i++) { var c = parseOA(items[i]); if (verifyMatch(c, origAuthor, origYear)) { oaMeta = c; break; } } } } catch(e) { /* silent */ } try { var crUrl = 'https://api.crossref.org/works?query.bibliographic=' + encodeURIComponent(searchTitle) + '&rows=5&mailto=techeconomist@research.org'; var crRes = UrlFetchApp.fetch(crUrl, { muteHttpExceptions: true }); if (crRes.getResponseCode() === 200) { var crData = JSON.parse(crRes.getContentText()); var items2 = (crData.message && crData.message.items) ? crData.message.items : []; for (var j = 0; j < items2.length; j++) { var c2 = parseCR(items2[j]); if (verifyMatch(c2, origAuthor, origYear)) { crMeta = c2; break; } } } } catch(e) { /* silent */ } if (oaMeta || crMeta) { if (!oaMeta) return crMeta; if (!crMeta) return oaMeta; return scoreResult(oaMeta) >= scoreResult(crMeta) ? oaMeta : crMeta; } // Strategy 3: Fall back to parsing the reference text directly return parseRefText(refText); }
